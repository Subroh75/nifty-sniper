name: Nifty Sniper Morning Sync
on:
  schedule:
    - cron: '45 3 * * 1-5' # 9:15 AM IST
  workflow_dispatch:      # The "Run" button

jobs:
  run-sniper:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout Code
        uses: actions/checkout@v4

      - name: Setup Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.10'

      - name: Install Libs
        run: pip install pandas yfinance gspread oauth2client

      - name: Execute Script
        env:
          GCP_SERVICE_ACCOUNT: ${{ secrets.GCP_SERVICE_ACCOUNT }}
          GSHEET_NAME: ${{ secrets.GSHEET_NAME }}
        run: python morning_sync.py
